static int tauShouldFilterTest(const char* filter, const char* testcase);
static int tauShouldFilterTest(const char* filter, const char* testcase) {
    if(filter) {
        const char* filter_curr = filter;
        const char* testcase_curr = testcase;
        const char* filter_wildcard = ((void*)0);

        while ((*filter_curr != '\0') && (*testcase_curr != '\0')) {
            if('*' == *filter_curr) {

                filter_wildcard = filter_curr;


                filter_curr++;

                while ((*filter_curr != '\0') && (*testcase_curr != '\0')) {
                    if(*filter_curr == '*') {


                        break;
                    } else if(*filter_curr != *testcase_curr) {

                        filter_curr = filter_wildcard;
                    }


                    testcase_curr++;


                    filter_curr++;
                }

                if((*filter_curr == '\0') && (*testcase_curr == '\0'))
                    return 0;


                if(*testcase_curr == '\0')
                    return 1;
            } else {
                if(*testcase_curr != *filter_curr) {

                    return 1;
                } else {

                    testcase_curr++;
                    filter_curr++;
                }
            }
        }

        if((*filter_curr != '\0') || ((*testcase_curr == '\0') && ((filter == filter_curr) ||
            (filter_curr[-1] != '*')))) {

            return 1;
        }
    }
    return 0;
}
